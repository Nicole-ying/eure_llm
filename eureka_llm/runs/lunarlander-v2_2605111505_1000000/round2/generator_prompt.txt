    dist_x = abs(state[0])

    r_alive = 0.01

    components = {
        "r_stability": r_stability,
        "r_proximity": r_proximity,
        "r_legs": r_legs,
        "r_alive": r_alive,
    }

    # Total before terminal outcome (sum of all dense components)
    total = r_stability + r_proximity + r_legs + r_alive
