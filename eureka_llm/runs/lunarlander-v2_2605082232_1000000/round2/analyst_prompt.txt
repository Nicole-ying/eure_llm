    # and provides a strong per-step incentive to stay alive.
    r_alive = 0.5   # Constant survival bonus to incentivize staying alive regardless of distance

    # Bonus for both legs in contact (indicates landing attempt)
    r_leg_contact = 1.0 if (state[6] >= 0.5 and state[7] >= 0.5) else 0.0

    # Penalty for vertical speed (downward = positive in state[3], so abs)
    # 0.05 * 7 ≈ 0.35, keeps penalty moderate while discouraging fast falling
    r_vertical_speed = -0.05 * abs(state[3])   # state[3] is vertical speed; penalizes fast falling

    # Compute total BEFORE adding _outcome
    components = {
        "r_distance": r_distance,
        "r_alive": r_alive,
        "r_leg_contact": r_leg_contact,
        "r_vertical_speed": r_vertical_speed,
    }
    total = sum(components.values())

        components["_outcome"] = 1.0 if success else -1.0  # not added to total
    # else: do NOT add _outcome key at all

    return total, components


def metrics_fn(env, action) -> dict:
